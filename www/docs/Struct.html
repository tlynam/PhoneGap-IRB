def _inspect
  str = "#<struct #{self.class.to_s} "
  buf = []
  self.each_pair do |k,v|
    buf.push [k.to_s + "=" + v._inspect]
  end
  str + buf.join(", ") + ">"
end

def select(&block)
  ary = []
  self.class.members.each{|field|
    val = self[field]
    ary.push(val) if block.call(val)
  }
  ary
end
